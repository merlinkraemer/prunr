  <action>Create a new dropdown menu component in MenuBarView:

```swift
private var pathDropdownMenu: some View {
    Menu {
        ForEach(Array(SettingsStore.shared.trackedPaths.enumerated()), id: \.element.id) { index, path in
            Button(action: {
                selectedPathIndex = index
                // Update manager to use selected path
                manager.switchToPath(path)
            }) {
                HStack {
                    if index == selectedPathIndex {
                        Image(systemName: "checkmark")
                            .foregroundStyle(.blue)
                    }
                    Text(path.url.lastPathComponent)
                        .foregroundStyle(.primary)
                    Spacer()
                    Text(formattedBytes(manager.pathSize(for: path)))
                        .font(.system(size: 11, weight: .medium))
                        .foregroundStyle(.secondary)
                        .monospaced()
                }
            }
            .disabled(index == selectedPathIndex)
        }
    } label: {
        HStack(spacing: 6) {
            Image(systemName: "chevron.down")
                .font(.system(size: 9, weight: .medium))
                .foregroundStyle(.secondary)
                .rotationEffect(isShowingDropdown ? .degrees(180) : .degrees(0))
                .animation(.easeInOut(duration: 0.2), value: isShowingDropdown)
        }
        .padding(.horizontal, 6)
        .padding(.vertical, 2)
        .background(
            Capsule()
                .fill(Color.gray.opacity(0.15))
        )
    }
    .menuStyle(.borderlessButton)
    .onTapGesture {
        withAnimation(.easeInOut(duration: 0.2)) {
            isShowingDropdown.toggle()
        }
    }
    .onAppear {
        // Initialize selected path index
        if let index = SettingsStore.shared.trackedPaths.firstIndex(where: { $0.id == manager.currentPathId }) {
            selectedPathIndex = index
        }
    }
}
```

Add this component after the helper methods.</action>
